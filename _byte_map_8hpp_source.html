56  template <class It, typename = std::enable_if_t<std::is_convertible_v<
57  decltype(*std::declval<It>()), std::byte>>>
58  bool setData(Addr A, boost::iterator_range<It> Data) {
59  // Look for a region to hold this data. If necessary, extend or merge
60  // existing regions to keep allocations contiguous.
61  auto data_size = std::distance(Data.begin(), Data.end());
62  Addr Limit = A + data_size;
63  for (size_t i = 0; i < Regions.size(); i++) {
64  auto& Current = Regions[i];
65 
66  // Overwrite data in existing region
67  if (containsAddr(Current, A) && Limit <= addressLimit(Current)) {
68  auto Offset = A - Current.Address;
69  std::copy(Data.begin(), Data.end(), Current.Data.begin() + Offset);
70  return true;
71  }
72 
73  // Extend region
74  if (A == addressLimit(Current)) {
75  bool HasNext = i + 1 < Regions.size();
76  if (HasNext && Limit > Regions[i + 1].Address) {
77  return false;
78  }
79 
80  Current.Data.reserve(Current.Data.size() + data_size);
81  std::copy(Data.begin(), Data.end(), std::back_inserter(Current.Data));
82  // Merge with subsequent region
83  if (HasNext && Limit == Regions[i + 1].Address) {
84  const auto& D = Regions[i + 1].Data;
85  Current.Data.reserve(Current.Data.size() + D.size());
86  std::copy(D.begin(), D.end(), std::back_inserter(Current.Data));
87  this->Regions.erase(this->Regions.begin() + i + 1);
88  }
89  return true;
90  }
91 
92  // Extend region backward
93  if (Limit == Current.Address) {
94  // Note: this is probably O(N^2), moving existing data on each inserted
95  // element.
96  std::copy(Data.begin(), Data.end(),
97  std::inserter(Current.Data, Current.Data.begin()));
98  Current.Address = A;
99  return true;
100  }
101 
102  if (containsAddr(Current, A) || containsAddr(Current, Limit - 1)) {
103  return false;
104  }
105  }
106 
107  // Not contiguous with any existing data. Create a new region.
108  Region R = {A, std::vector<std::byte>()};
109  R.Data.reserve(data_size);
110  std::copy(Data.begin(), Data.end(), std::back_inserter(R.Data));
111  this->Regions.insert(
112  std::lower_bound(this->Regions.begin(), this->Regions.end(), R,
113  [](const auto& Left, const auto& Right) {
114  return Left.Address < Right.Address;
115  }),
116  std::move(R));
117 
118  return true;
119  }

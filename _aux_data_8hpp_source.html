163 // Serialize and deserialize by copying the object representation directly.
164 template <class T> struct default_serialization {
165  static void toBytes(const T& object, to_iterator It) {
166  // Store as little-endian.
167  T reversed = boost::endian::conditional_reverse<
168  boost::endian::order::little, boost::endian::order::native>(object);
169  auto srcBytes_begin = reinterpret_cast<std::byte*>(&reversed);
170  auto srcBytes_end = reinterpret_cast<std::byte*>(&reversed + 1);
171  std::transform(srcBytes_begin, srcBytes_end, It,
172  [](auto b) { return char(b); });
173  }
174 
175  static from_iterator fromBytes(T& object, from_iterator It) {
176  auto dest_begin = reinterpret_cast<std::byte*>(&object);
177  auto dest_end = reinterpret_cast<std::byte*>(&object + 1);
178  std::for_each(dest_begin, dest_end, [&It](auto& b) {
179  b = std::byte(*It);
180  ++It;
181  });
182  // Data stored as little-endian.
183  boost::endian::conditional_reverse_inplace<boost::endian::order::little,
184  boost::endian::order::native>(
185  object);
186  return It;
187  }
188 };

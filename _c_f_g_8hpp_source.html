75 // Helper for constructing the CFG type. The graph property needs to refer to
76 // the graph's vertex_descriptor type. This is accessible via
77 // boost::adjacency_list_traits, but requires keeping the template parameters
78 // for boost::adjacency_list and boost::adjacency_list_traits in sync. This
79 // helper ensures the relevant parameters are the same for both.
80 template <class OutEdgeListS = boost::vecS, class VertexListS = boost::vecS,
81  class DirectedS = boost::directedS, class EdgeListS = boost::listS>
82 struct CfgBuilder {
83  using vertex_descriptor = typename boost::adjacency_list_traits<
84  OutEdgeListS, VertexListS, DirectedS, EdgeListS>::vertex_descriptor;
85  using type = boost::adjacency_list<
86  OutEdgeListS, VertexListS, DirectedS,
87  // Vertices are CfgNodes.
88  CfgNode*,
89  // Edges have labels.
90  EdgeLabel,
91  // The graph keeps track of vertex descriptors for
92  // each node.
93  std::unordered_map<const CfgNode*, vertex_descriptor>, EdgeListS>;
94 };

39  auto visit_impl_help(Callable&& Visitor) const
40  -> std::enable_if_t<std::is_void_v<CommonRetTy>, bool> {
41  static_assert(std::is_invocable_v<Callable, Ty*>,
42  "Visitor must contain an overloaded function call operator "

71  // Helper type traits used to determine whether all of the Callable object's
72  // return types agree. Note that the return types do not need to be the same,
73  // but do need to be implicitly convertible to the same common type. e.g.,

76  // int operator()(Block*);

103  // Helper function to unpack all of the types in the type list and attempt to
104  // visit Callable once per type. Verifies that the Callable objects all share
105  // a compatible return type.
106  //
107  // SFINAE overload handling a Callable where there is a common, non-void
108  // return type. Returns a std::optional<common_type>.
109  template <typename Callable, template <typename...> typename TypeList,
110  typename... Types>
111  auto visit_impl(Callable&& Visitor, TypeList<Types...>) const
112  -> std::enable_if_t<
113  !std::is_void_v<common_return_type_t<Callable, TypeList, Types...>>,
114  std::optional<common_return_type_t<Callable, TypeList, Types...>>> {
115  // If this assertion fails, the return values from the Callable object are
116  // not compatible enough. This can happen if they return incompatible types
117  // or if there is not an overload for each referent type.
118  static_assert(common_return_type_v<Callable, TypeList, Types...>,
119  "incompatible return types for the Callable object");
120  // Instantiate a call to the Visitor once for each of the listed types, but
121  // only issue the call at runtime if the Referent can be dynamically cast to
122  // the given type. In this way, the Visitor needs to be able to handle any
123  // of the supported types, but will only be called once for the concrete
124  // type of the Referent.
125  //
126  // If you get an error about there being no matching overloaded function for
127  // the call to visit_impl_help, that is most likely because there are one
128  // or more overloads missing for each referent type.
129  std::optional<common_return_type_t<Callable, TypeList, Types...>> Res;
130  (... ||
131  visit_impl_help<Callable, Types, typename decltype(Res)::value_type>(
132  std::forward<Callable>(Visitor), Res));
133  return Res;
134  }
135 
136  // SFINAE overload handling a Callable where every return type is void.
137  // Returns void.
138  template <typename Callable, template <typename...> typename TypeList,
139  typename... Types>
140  auto visit_impl(Callable&& Visitor, TypeList<Types...>) const
141  -> std::enable_if_t<
142  std::is_void_v<common_return_type_t<Callable, TypeList, Types...>>,
143  void> {
144  // Call each of the overloads on Callable, but there is no value to be
145  // returned from any of the calls.
146  (... ||
147  visit_impl_help<Callable, Types, void>(std::forward<Callable>(Visitor)));
148  }

271  std::optional<Addr> getAddress() const {
272  return std::visit(
273  [](const auto& arg) {
274  using T = std::decay_t<decltype(arg)>;
275  if constexpr (std::is_same_v<T, std::monostate>)
276  return std::optional<Addr>{};

281  return std::make_optional(b->getAddress());
